USE `ho`;
DROP function IF EXISTS `get_coordinates_id`;

DELIMITER $$
USE `ho`$$
CREATE FUNCTION `get_coordinates_id` (platitude INT,plongitude INT)
RETURNS INTEGER
DETERMINISTIC
BEGIN
DECLARE vcName INT;
SET vcName = (SELECT id_coordinates FROM coordinates WHERE latitude = platitude AND longitude = plongitude);
RETURN vcName;
END$$

DELIMITER ;
