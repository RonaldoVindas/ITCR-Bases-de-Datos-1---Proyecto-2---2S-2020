USE `ho`;
DROP procedure IF EXISTS `insert_person`;

DELIMITER $$
USE `ho`$$
CREATE PROCEDURE `insert_person` (IN pfirst_name VARCHAR(45),IN plast_name VARCHAR(45),IN pbirthday DATE,IN pcommunity VARCHAR(100),IN pemail VARCHAR(45),IN pusername VARCHAR(45),IN pbank_account INT,IN pid_gender INT,IN pid_nationality INT,IN pid_user_type INT)
BEGIN
DECLARE EXIT HANDLER FOR 1062 SELECT 'Duplicate keys error encountered.' Message;
DECLARE EXIT HANDLER FOR 1105 SELECT 'Unknown error encountered.' Message;
DECLARE EXIT HANDLER FOR 1172 SELECT 'Result consisted of more than one row.' Message;
DECLARE EXIT HANDLER FOR 1176 SELECT 'Key does not exist.' Message;
DECLARE EXIT HANDLER FOR 1231 SELECT 'Variable cannot be set to that value.' Message;
DECLARE EXIT HANDLER FOR 1232 SELECT 'Incorrect argument type to variable.' Message;
	INSERT INTO person(first_name,last_name,birthday,community,email,username,bank_account,id_gender,id_nationality,id_user_type)
    VALUES(pfirst_name,plast_name,pbirthday,pcommunity,pemail,pusername,pbank_account,pid_gender,pid_nationality,pid_user_type);
END$$

DELIMITER ;
